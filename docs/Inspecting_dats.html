plot_pca_dataset <- function(data, color_by = c("Group", "Experiment"), title_suffix = "") {
  color_by <- match.arg(color_by)
  
  # 1. Pivot data wide (Samples as rows, Assays as columns)
  npx_wide <- data |>
    dplyr::select(SampleID, OlinkID, NPX) |>
    tidyr::pivot_wider(
      names_from = OlinkID, 
      values_from = NPX, 
      values_fn = mean
    ) |>
    tibble::column_to_rownames("SampleID")
  
  # 2. Drop assays with any NA across samples
  npx_wide_clean <- npx_wide[, colSums(is.na(npx_wide)) == 0]
  n_assays <- ncol(npx_wide_clean)
  
  # 3. Compute PCA
  pca_res <- prcomp(npx_wide_clean, scale. = TRUE)
  var_explained <- summary(pca_res)$importance[2, 1:2] * 100
  pc1_lab <- paste0("PC1 (", round(var_explained[1], 1), "%)")
  pc2_lab <- paste0("PC2 (", round(var_explained[2], 1), "%)")
  
  # 4. Extract metadata & join PCA coordinates
  meta_df <- data |>
    dplyr::select(SampleID, Group, Experiment) |>
    dplyr::distinct()
  
  pca_df <- as.data.frame(pca_res$x[, 1:2]) |>
    tibble::rownames_to_column("SampleID") |>
    dplyr::left_join(meta_df, by = "SampleID")
  
  # 5. Setup plot configuration based on coloring factor
  if (color_by == "Group") {
    p_color_aes <- aes(color = Group)
    p_scale     <- scale_color_manual(values = custom_col5)
    color_title <- "Experimental Group"
  } else {
    p_color_aes <- aes(color = Experiment)
    p_scale     <- scale_color_manual(values = colors4)
    color_title <- "Experiment Batch"
  }
  
  # 6. Generate Plot
  ggplot2::ggplot(pca_df, aes(x = PC1, y = PC2)) +
    ggplot2::geom_point(p_color_aes, size = 4, alpha = 0.85) +
    ggrepel::geom_text_repel(
      aes(label = SampleID), 
      size = 3, 
      show.legend = FALSE, 
      max.overlaps = 15
    ) +
    p_scale +
    ggplot2::labs(
      title = paste0("PCA: ", title_suffix),
      subtitle = paste0("Based on ", n_assays, " NPX protein profiles"),
      x = pc1_lab,
      y = pc2_lab,
      color = color_title
    ) +
    ggplot2::theme_bw(base_size = 12) +
    ggplot2::theme(
      plot.title = ggplot2::element_text(face = "bold", size = 12),
      plot.subtitle = ggplot2::element_text(face = "italic", size = 10),
      legend.position = "right",
      panel.grid.minor = ggplot2::element_blank()
    )
}
