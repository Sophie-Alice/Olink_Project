# NPX data into matrix format (Proteins = Rows, Samples = Columns)
npx_mat <- filt.NPX |>
  dplyr::select(SampleID, Assay, NPX) |>
  tidyr::pivot_wider(names_from = SampleID, values_from = NPX, values_fn = mean) |>
  tibble::column_to_rownames("Assay") |>
  as.matrix()

# Extract sample metadata matching the matrix column order
sample_meta <- filt.NPX |>
  dplyr::select(SampleID, Group, Experiment) |>
  dplyr::distinct() |>
  dplyr::slice(match(colnames(npx_mat), SampleID))

# Clean group names for limma syntax (removes spaces and special characters)
clean_group <- factor(make.names(sample_meta$Group))

# Create Design Matrix
design <- model.matrix(~ 0 + clean_group)
colnames(design) <- levels(clean_group)


#############################################

# Fit initial linear model across all proteins
fit <- lmFit(npx_mat, design)

# Define contrasts
contrast_matrix <- makeContrasts(
  # 1. Macrophage effect under TGFb
  HM_TGF_vs_NC_TGF = MT...Macrophages..TGFb. - MT.Control..TGFb.,
  # 2. TGFb effect in MT without macrophages
  NC_TGF_vs_NC_NC  = MT.Control..TGFb. - MT.Control..NC.,
  # 3. Macrophage effect at baseline (NC)
  HM_NC_vs_NC_NC   = MT...Macrophages..NC. - MT.Control..NC.,
  # 4. TGFb effect with Macrophages present
  HM_TGF_vs_HM_NC  = MT...Macrophages..TGFb. - MT...Macrophages..NC.,
  # 5. The Interaction term = How does Mac presence affect the response to TGFb?
  Interaction      = (MT...Macrophages..TGFb. - MT...Macrophages..NC.) - 
                     (MT.Control..TGFb. - MT.Control..NC.),
  levels = design
)

# Fit contrasts & apply Empirical Bayes moderation
fit_contrasts <- contrasts.fit(fit, contrast_matrix)
fit_ebayes    <- dBayes <- eBayes(fit_contrasts)

#############################################


# Create empty list to hold tables
limma_results_list <- list()

# Loop through all 5 contrast names and store tables
for (c_name in colnames(contrast_matrix)) {
  
  res_table <- topTable(fit_ebayes, coef = c_name, number = Inf, adjust.method = "BH") |>
    tibble::rownames_to_column("OlinkID") |>
    dplyr::mutate(
      Significance = dplyr::case_when(
        adj.P.Val < 0.05 & logFC > 0.5  ~ "Up",
        adj.P.Val < 0.05 & logFC < -0.5 ~ "Down",
        TRUE                           ~ "Not Sig"
      )
    )
  
  # Add to named list
  limma_results_list[[c_name]] <- res_table
}

##################################### Function of Hallmark fGSEA

run_gsea_pipeline <- function(degs_df, comparison_name, pathways = Hallmark.pathways, top_n = 15) {
  
  # Clean data & create ranked named vector
  protein_ranked_df <- degs_df %>%
    dplyr::select(OlinkID, logFC) %>% 
    na.omit() %>%
    distinct() %>%
    group_by(OlinkID) %>%
    summarize(log2FC = mean(logFC, na.rm = TRUE)) %>%
    arrange(desc(log2FC))
  
  ranked_proteins <- deframe(protein_ranked_df)
  
  # Run GSEA (Using fgseaMultilevel)
  fgseares <- fgseaMultilevel(pathways = pathways, stats = ranked_proteins)
  
  # Process Tidy Results
  fgseares.tidy <- fgseares %>%
    as_tibble() %>%  
    arrange(desc(NES))
  
  # Extract Top Up and Down pathways
  fgseares.tidy.top.up   <- head(fgseares.tidy, top_n)
  fgseares.tidy.top.down <- tail(fgseares.tidy, top_n)
  fgseares.tidy.top.both <- bind_rows(fgseares.tidy.top.up, fgseares.tidy.top.down)
  
  # Format dataframe for plotting
  plot_df <- fgseares.tidy.top.both %>%
    mutate(
      padj_group = case_when(
        padj <= 0.05 ~ "< 0.05",
        padj <= 0.2  ~ "0.05 – 0.2",
        TRUE         ~ "> 0.2"
      ),
      padj_group = factor(padj_group, levels = c("< 0.05", "0.05 – 0.2", "> 0.2")),
      pathway = gsub("HALLMARK_", "", pathway)
    )
  
  # Generate Plot
  p <- ggplot(plot_df, aes(reorder(pathway, NES), NES)) +
    geom_col(aes(fill = padj_group), color = "black", linewidth = 0.2) +
    coord_flip() +
    scale_x_discrete(position = "top", labels = scales::label_wrap(50)) +
    scale_fill_manual(
      name = "adj. p-value",
      values = c(
        "< 0.05"     = "#9F1C67",   
        "0.05 – 0.2" = "#FFD966",   
        "> 0.2"      = "#005EA8"    
      ),
      drop = FALSE 
    ) +
    labs(
      x = "", 
      y = "Normalized Enrichment Score (NES)", 
      title = "GSEA: Hallmark Pathways",
      subtitle = paste("Comparison:", comparison_name)
    ) +
    theme_bw() +
    theme(
      aspect.ratio    = 1.4, 
      axis.text.y     = element_text(size = 8, hjust = 0), 
      axis.title      = element_text(face = "bold"),
      plot.title      = element_text(hjust = 0, face = "bold"),
      plot.subtitle   = element_text(hjust = 0, size = 9, face = "italic"),
      legend.position = "left" 
    )
  
  # Return both table and plot in a list
  return(list(
    results_table = fgseares.tidy,
    plot          = p
  ))
}
